    \begin{itemize}
        \item \textbf{Requirement Being Tested:} R1 – Register to access personalized features

        \item \textbf{Preconditions:}
        \begin{itemize}
            \item System is running
            \item Evaluator has no existing account  for the test email.
        \end{itemize}
        \item \textbf{Test Steps:}
        \begin{enumerate}
            \item Open the application.
            \item Navigate to Sign Up / Register.
            \item Enter valid registration data (username/email/password as required).
            \item Submit the registration form.
            \item Try logging in with the newly created credentials.

        \item \textbf{Actual Result:} PASS
        \item \textbf{Comments:}
        \begin{itemize}
            \item No email verification and no reset/ forgot password. 
        \end{itemize}

    \item \textbf{Acceptance Test Case 05}

    \begin{itemize}
        \item \textbf{Requirement Being Tested:} R3 – Record trip in Manual Mode
        \item \textbf{Preconditions:}
        \begin{itemize}
            \item User is logged in.
        \end{itemize}
        \item \textbf{Test Steps:}
        \begin{enumerate}
            \item Navigate to Create Trip (Manual mode).
            \item Enter trip metadata (name/date/time/duration as required).
            \item Enter street/path info and select a status for the path/segments.
            \item Save trip.
        \end{enumerate}
        \item \textbf{Expected Result:} Trip is saved; user can view it in their trip history with entered data.
        \item \textbf{Actual Result:} PASS
        \item \textbf{Comments:}
        \begin{itemize}
            \item Unable to edit the entered points/ locations. 
            \item Destination limits in the dropdown for “End Point”. 
            \item The dropdown has only 3-4 places listed for "End Point". 
            \item The start and end points are predetermined. No choice is given for users to search for places. 
            \item No option to edit/delete the entered path. 
            \item Unable to edit the manually saved path.
        \end{itemize}
    \end{itemize}

    \item \textbf{Acceptance Test Case 20}
    \begin{itemize}
        \item \textbf{Requirement Being Tested:}  R16 – Multi-user merge: majority wins
        \item \textbf{Preconditions:}
        \begin{itemize}
            \item Two (or more) different users exist 
            \item Publishing is available 
            \item Both can report on the same segment/path.
        \end{itemize}
        \item \textbf{Test Steps:}
        \begin{enumerate}
            \item User A creates a trip on segment S and sets status = optimal; publish.
            \item User B creates a trip on the same segment S and sets status = maintenance; publish.
            \item User C (or A/B) creates a third published report on the same segment S = optimal.
            \item Run a search that includes segment S or view aggregated segment status.
        \end{enumerate}
        \item \textbf{Expected Result:} Aggregated/merged status for S becomes “optimal” (majority).
        \item \textbf{Actual Result:} Not Executed
        \item \textbf{Comments:}
        \begin{itemize}
            \item Not able to verify the merge whether it is working or not.
        \end{itemize}
    \end{itemize}
    \item \textbf{Acceptance Test Case 21}

    \begin{itemize}
        \item \textbf{Requirement Being Tested:}  R16 – Merge tie-breaker uses most recent timestamp
        \item \textbf{Preconditions:}
        \begin{itemize}
            \item Two users are needed.
            \item Equal number of reports for two different statuses on same segment/path.
        \end{itemize}
        \item \textbf{Test Steps:}
        \begin{enumerate}
            \item User A publishes status for segment S = medium.
            \item User B publishes status for segment S = sufficient (same count now).
            \item Publish a newer report to break tie (e.g., User B publishes again same segment S = sufficient later) OR follow the system’s mechanism to update.
            \item Observe merged outcome for S.
        \end{enumerate}
        \item \textbf{Expected Result:} When counts tie, the system selects the most recent report’s status.
        \item \textbf{Actual Result:} PASS
        \item \textbf{Comments:}
        \begin{itemize}
            \item It counts ties, but displaying both the paths and not able to verify which is considered.
        \end{itemize}
    \end{itemize}
